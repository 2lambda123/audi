152  auto phat2 = phat * phat;
153

155  double cos_p0 = std::cos(p0);
156 
157  double factorial=1.;
158  double coeff=1.;
159  gdual cos_taylor(1., d.get_order());
160  gdual tmp(cos_taylor);
161  for (auto i=2; i<=d.get_order(); i+=2) {
162  coeff*=-1.; // -1, 1, -1, 1, ...
163  tmp*=phat2; // phat^2, phat^4, phat^6 ...
164  factorial*=i * (i-1); // 2!, 4!, 6!, ...
165  cos_taylor += (coeff/factorial) * tmp;
166  }

168  factorial=1.;
169  coeff=1.;
170  gdual sin_taylor(phat);

173  coeff*=-1.; // -1, 1, -1, 1, ...

175  factorial*=i * (i-1); // 3!, 5!, 7!, ...
176  sin_taylor += (coeff/factorial) * tmp;

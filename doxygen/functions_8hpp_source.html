292  auto phat2 = phat * phat;

300  gdual tmp(cos_taylor);

302  coeff*=-1.; // -1, 1, -1, 1, ...
303  tmp*=phat2; // phat^2, phat^4, phat^6 ...
304  factorial*=i * (i-1); // 2!, 4!, 6!, ...
305  cos_taylor += (coeff/factorial) * tmp;
306  }

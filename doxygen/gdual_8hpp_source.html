29 // Multiplier functor for the the gdual class, based on
30 // Piranha's polynomial multiplier.
31 class gdual_multiplier: piranha::series_multiplier<p_type>
32 {
33  using base = piranha::series_multiplier<p_type>;
34  public:
35  explicit gdual_multiplier(const p_type &p1, const p_type &p2, int max_degree):base(p1,p2),m_max_degree(max_degree) {}
36  p_type operator()() const
37  {
38  using term_type = p_type::term_type;
39  using degree_type = piranha::integer;
40  using size_type = base::size_type;
41  // First let's order the terms in the second series according to the degree.
42  std::stable_sort(this->m_v2.begin(),this->m_v2.end(),[this](term_type const *p1, term_type const *p2) {
43  return p1->m_key.degree(this->m_ss) < p2->m_key.degree(this->m_ss);
44  });
45  // Next we create two vectors with the degrees of the terms in the two series. In the second series,
46  // we negate and add the max degree in order to avoid adding in the skipping functor.
47  std::vector<degree_type> v_d1, v_d2;
48  std::transform(this->m_v1.begin(),this->m_v1.end(),std::back_inserter(v_d1),[this](term_type const *p) {
49  return p->m_key.degree(this->m_ss);
50  });
51  std::transform(this->m_v2.begin(),this->m_v2.end(),std::back_inserter(v_d2),[this](term_type const *p) {
52  return this->m_max_degree - p->m_key.degree(this->m_ss);
53  });
54  // The skipping functor.
55  auto sf = [&v_d1,&v_d2](const size_type &i, const size_type &j) -> bool {
56  using d_size_type = decltype(v_d1.size());
57  return v_d1[static_cast<d_size_type>(i)] > v_d2[static_cast<d_size_type>(j)];
58  };
59  // The filter functor: will return 1 if the degree of the term resulting from the multiplication of i and j
60  // is greater than the max degree, zero otherwise.
61  auto ff = [&sf](const size_type &i, const size_type &j) {
62  return static_cast<unsigned>(sf(i,j));
63  };
64  return this->plain_multiplication(sf,ff);
65  }
66  private:
67  const int m_max_degree;
68 };

184  // Dual * dual.
185  static gdual mul(const gdual &d1, const gdual &d2)
186  {
187  const int order = d1.get_order();
188  if (order != d2.get_order()) {
189  throw std::invalid_argument("different truncation limit");
190  }
191  const auto &ss1 = d1.m_p.get_symbol_set(), &ss2 = d2.m_p.get_symbol_set();
192  if (ss1 == ss2) {
193  return mul_impl(d1.m_p,d2.m_p,order);
194  }
195  // If the symbol sets are not the same, we need to merge them and make
196  // copies of the original operands as needed.
197  auto merge = ss1.merge(ss2);
198  const bool need_copy_1 = (merge != ss1), need_copy_2 = (merge != ss2);
199  if (need_copy_1) {
200  p_type copy_1(d1.m_p.extend_symbol_set(merge));
201  if (need_copy_2) {
202  p_type copy_2(d2.m_p.extend_symbol_set(merge));
203  return mul_impl(copy_1,copy_2,order);
204  }
205  return mul_impl(copy_1,d2.m_p,order);
206  } else {
207  p_type copy_2(d2.m_p.extend_symbol_set(merge));
208  return mul_impl(d1.m_p, copy_2,order);
209  }
210  }

224  // Basic overloads for the division
225  static gdual div(const gdual &d1, const gdual &d2)
226  {
227  if (d1.get_order() != d2.get_order()) {
228  throw std::invalid_argument("different truncation limit");
229  }
230 
231  gdual retval(1, d2.get_order());
232  double fatt = -1;
233  auto p0 = d2.constant_cf();
234  if (p0 == 0) {
235  throw std::domain_error("divide by zero");
236  }
237  auto phat = (d2 - p0);
238  phat = phat/p0;
239  gdual tmp(phat);
240 
241  retval = retval - phat;
242  for (auto i = 2; i <= d2.m_order; ++i) {
243  fatt *= -1;
244  phat*=tmp;
245  retval = retval + fatt * phat;
246  }
247 
248  return (d1 * retval) / p0;
249  }
